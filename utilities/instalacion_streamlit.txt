### Abre el firewall en IONOS

Campo

### Ejecutar
cd Dashboard
streamlit run app_estorninos.py --server.port 8501 --server.address 0.0.0.0

#El --server.address 0.0.0.0 es importante para que sea accesible desde fuera del servidor.

### Crea un servicio para que arranque automáticamente:
sudo nano /etc/systemd/system/estorninos.service

[Unit]
Description=Estorninos Dashboard
After=network.target

[Service]
User=root
WorkingDirectory=/root/Dashboard
ExecStart=/root/.env/streamlit/bin/streamlit run run_estorninos.py \
          --server.port 8501 \
          --server.address 0.0.0.0 \
          --server.headless true
Restart=always
RestartSec=5
StandardOutput=append:/root/Dashboard/log/estorninos.log
StandardError=append:/root/Dashboard/log/estorninos.log

[Install]
WantedBy=multi-user.target

### Para que el log no crezca indefinidamente
sudo nano /etc/logrotate.d/estorninos
/root/Dashboard/estorninos.log {
    weekly
    rotate 4
    compress
    missingok
    notifempty
}

### Activarlo:
sudo systemctl daemon-reload
sudo systemctl enable estorninos
sudo systemctl start estorninos
sudo systemctl status estorninos

# Parar el servicio
sudo systemctl stop estorninos

# Reiniciar el servicio
sudo systemctl restart estorninos

# Ver logs en tiempo real
tail -f /root/Dashboard/estorninos.log
